				+ '/us/emailsignup/etemailsignupform';
				
		if(!cancel){
			var dat = {emailId : emailIdVal ,subscriptionItems : $("#emailGroupsUID").val().split(";")};
		}
		else{
			var dat = {emailId : emailIdVal ,subscriptionItems :["Cancel"]};
		}

		dat = JSON.stringify(dat);

		$.ajax({
			//url : "/us/emailsignup/etemailsignupform",
			url: strUrl,
			type : 'POST',
			headers: { 
				 'Accept': 'application/json',
				 'Content-Type': 'application/json' 
			},
			data : dat,
			success : function(data) {
				$('.submit-msg').css('display', 'none');
				$('.pp_content_container fieldset#body #emailsignup2').css('display', 'inline');

				if(!cancel) {
					$("#email-signup").html($(".pp_content_container").html());
					if (data.success){
						$(".pp_content_container").html("<H2>" + thankyouMsg + "</H2>");
					}
					else {
						$(".pp_content_container").html("<H2>" + thankyouMsg + "</H2>");
					}
					resizeContainer();
				}
			}
		});
	}
	return false;
}

function findSelPrefs(){
	var allCodes = "";
	var firstEle = true;
	$("input[type=checkbox][class='footer form-checkbox']").each(function(index) {
		var code = $(this).attr("id");

			if(firstEle) {
				allCodes = code;
				firstEle = false;
			} else {
				allCodes = allCodes + ";" + code;
			}
		}
	});

	$("#emailGroupsUID").val(allCodes);
}
